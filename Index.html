# atm_simulator.py # Initialize balance balance = 0.0 while True: print("\nATM Simulator") print("1. Check Balance") print("2. Deposit") print("3. Withdraw") print("4. Exit") choice = input("Choose an option: ") if choice == '1': print(f"Your current balance is: ${balance:.2f}") elif choice == '2': amount = float(input("Enter amount to deposit: ")) if amount > 0: balance += amount print(f"Deposited: ${amount:.2f}") else: print("Deposit amount must be positive.") elif choice == '3': amount = float(input("Enter amount to withdraw: ")) if 0 < amount <= balance: balance -= amount print(f"Withdrew: ${amount:.2f}") else: print("Insufficient funds or invalid amount.") elif choice == '4': print("Thank you for using the ATM. Goodbye!") break else: print("Invalid choice. Please try again.")
